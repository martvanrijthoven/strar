[docs]class Text:
    BOLD = "\033[1m"
    BOLD_END = "\033[0m"

    @staticmethod
    def lower(text: str) -> str:
        return text.lower()

    @staticmethod
    def split_capitals_with_underscore(text: str) -> str:
        return "_".join(re.findall(".[^A-Z]*", text))
